“我的第一篇文章”

– 订单表
CREATE TABLE order_info (
id bigint(20) NOT NULL AUTO_INCREMENT,
account_id bigint(20) NOT NULL COMMENT ‘账户ID’,
order_number varchar(50) NOT NULL COMMENT ‘订单编号，唯一，是系统生产’,
pay_end_time datetime DEFAULT NULL COMMENT ‘最晚支付时间’,
total_price double NOT NULL COMMENT ‘订单总价格’,
pay_status varchar(20) NOT NULL COMMENT ‘支付状态，PayEnum枚举’,
deleted tinyint(4) DEFAULT NULL COMMENT ‘删除状态:0未删除,1已删除’,
deleted_time datetime DEFAULT NULL COMMENT ‘删除时间’,
tenant_id bigint(22) NOT NULL COMMENT ‘租户ID’,
create_user_id bigint(22) NOT NULL COMMENT ‘创建人’,
update_user_id bigint(22) NOT NULL COMMENT ‘修改人’,
create_time datetime NOT NULL COMMENT ‘创建时间’,
update_time datetime NOT NULL COMMENT ‘修改时间’,
version int(11) NOT NULL DEFAULT ‘1’ COMMENT ‘版本号’,
payed_price double NOT NULL DEFAULT ‘0’ COMMENT ‘已付金额’,
balance double NOT NULL DEFAULT ‘0’ COMMENT ‘余额’,
deposit double DEFAULT NULL COMMENT ‘订金’,
process_type int(11) NOT NULL DEFAULT ‘1’ COMMENT ‘OrderProcessTypeEnum 订单流程类型’,
audit_status varchar(50) DEFAULT NULL COMMENT ‘订单审批状态，关联枚举类PayRefundFormAuditStatusEnum’,
audit_account_id bigint(20) DEFAULT NULL COMMENT ‘审批人的账号ID’,
is_need_audit int(11) DEFAULT ‘0’ COMMENT ‘是否需要审批。0：不需要，1：需要’,
audit_date datetime DEFAULT NULL COMMENT ‘审批时间’,
PRIMARY KEY (id) USING BTREE,
UNIQUE KEY uni_orderNumber (order_number)
) ENGINE=InnoDB AUTO_INCREMENT=0 DEFAULT CHARSET=utf8mb4 ROW_FORMAT=COMPACT COMMENT=’订单信息’；

注：total_price（订单总价格）是实际需要支付的价格；后续会加个订单原价格；balance（余额）是每支付一笔就累加，balance比total_price多的话后续退款需要用到；
OrderProcessTypeEnum包含：
FULL_PAYMENT(1,”全款”),
DEPOSIT(2,”订金”),
BALANCE_PAYMENT(3,”尾款”),;

deposit（订金）OrderProcessTypeEnum类型是订金是维护的字段；
PayEunm 支付状态枚举包含：
WAIT_PAY(“待支付”),
PAYED(“已付款”),
CLOSE(“已关闭”),
PARTIAL(“部分付款”),
;

订单审批状态枚举PayRefundFormAuditStatusEnum：
AUDIT(“待审核”),
PASS(“审核通过”),
REJECT(“审核驳回”),
RETRIVE(“已撤回”)
;

CREATE TABLE order_product (
id bigint(20) NOT NULL AUTO_INCREMENT,
order_id bigint(20) NOT NULL COMMENT ‘订单ID，表order_info’,

type_id bigint(20) NOT NULL COMMENT ‘商品类型id，表pro_product_type’,
sart_time datetime DEFAULT NULL COMMENT ‘开始时间’,
end_time datetime DEFAULT NULL COMMENT ‘结束时间’,
deposit double NOT NULL DEFAULT ‘0’ COMMENT ‘订金’,
deleted tinyint(4) DEFAULT NULL COMMENT ‘删除状态:0未删除,1已删除’,
deleted_time datetime DEFAULT NULL COMMENT ‘删除时间’,
tenant_id bigint(22) NOT NULL COMMENT ‘租户ID’,

update_user_id bigint(22) NOT NULL COMMENT ‘修改人’,
create_time datetime NOT NULL COMMENT ‘创建时间’,
update_time datetime NOT NULL COMMENT ‘修改时间’,
remain_number int(11) NOT NULL DEFAULT ‘0’ COMMENT ‘商品剩余数量’,

KEY orderId_index (order_id)
) ENGINE=InnoDB AUTO_INCREMENT=1529661291531137025 DEFAULT CHARSET=utf8mb4 ROW_FORMAT=COMPACT COMMENT=’订单商品信息’；

注：订单商品表和订单表是多对一的关系；price（单个商品价格）* number（商品数量） = order_indo(订单表) 的订单总价格时支付状态是已支付，否则是待支付或未支付；

CREATE TABLE order_accom_apply (
id bigint(20) NOT NULL AUTO_INCREMENT COMMENT ‘主键’,
deleted tinyint(4) DEFAULT NULL COMMENT ‘删除状态:0未删除,1已删除’,
deleted_time datetime DEFAULT NULL COMMENT ‘删除时间’,
tenant_id bigint(22) NOT NULL COMMENT ‘租户ID’,
create_user_id bigint(22) NOT NULL COMMENT ‘创建人’,
update_user_id bigint(22) NOT NULL COMMENT ‘修改人’,
create_time datetime NOT NULL COMMENT ‘创建时间’,
update_time datetime NOT NULL COMMENT ‘修改时间’,
version int(11) DEFAULT ‘1’ COMMENT ‘版本号’,
term_id bigint(20) NOT NULL COMMENT ‘学期ID’,
student_id bigint(20) NOT NULL COMMENT ‘学生ID’,
account_id bigint(20) NOT NULL COMMENT ‘账号ID’,
start_date datetime NOT NULL COMMENT ‘开始时间’,
end_date datetime NOT NULL COMMENT ‘结束时间’,
accom_type int(11) NOT NULL COMMENT ‘住宿申请类型：1：整学期，2：中途住宿，3：临时。关联枚举类AccomTypeEnum’,
all_amount double NOT NULL COMMENT ‘订单总金额, accom_amount+bed_amount’,
audit_status varchar(50) DEFAULT NULL COMMENT ‘审批状态’,
pay_status varchar(50) DEFAULT NULL COMMENT ‘支付状态’,
accom_amount double DEFAULT NULL COMMENT ‘住宿金额’,
is_buy_bed int(11) DEFAULT ‘0’ COMMENT ‘是否购买床上用品,0:否，1：是，关联枚举类BooleanEnum’,
bed_amount double NOT NULL COMMENT ‘床上用品金额’,
remark varchar(2000) DEFAULT NULL COMMENT ‘情况说明’,
apply_reason varchar(400) DEFAULT NULL COMMENT ‘申请理由’,
invite_id bigint(20) DEFAULT NULL COMMENT ‘邀请ID，关联ls_invite表’,
order_id bigint(20) DEFAULT NULL COMMENT ‘订单ID，关联order_info’,
refund_status varchar(50) DEFAULT NULL COMMENT ‘退款状态’,
is_work_date int(11) NOT NULL DEFAULT ‘1’ COMMENT ‘是否工作日，0：否，1：是，关联枚举类BooleanEnum’,
is_not_work_date int(11) NOT NULL DEFAULT ‘0’ COMMENT ‘是否非工作日，0:否，1：是，关联枚举类BooleanEnum’,
is_free int(11) DEFAULT ‘0’ COMMENT ‘是否免费。0:否，1:是，默认否，关联枚举类BooleanEnum’,
audit_text varchar(400) DEFAULT NULL COMMENT ‘审批意见’,
audit_teacher_id bigint(20) DEFAULT NULL COMMENT ‘审批老师ID’,
cc_teacher_ids varchar(100) DEFAULT NULL COMMENT ‘抄送老师ID，以英文逗号隔开’,
is_need_audit int(11) NOT NULL COMMENT ‘是否需要审批。0:不需要，1:需要’,
audit_date datetime DEFAULT NULL COMMENT ‘审批的日期’,
payed_time datetime DEFAULT NULL COMMENT ‘支付时间’,
PRIMARY KEY (id),
KEY ind_accom_order (order_id)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COMMENT=’住宿订单’；

注：关联 order_info 表的 order_id；具体义务具体设计

最后：下单的整个流程是：第一步，设置pro_product商品信息表、pro_product_type商品类型表。第二步，创建业务申请单order_accom_apply 完善业务申请单表的数据。第三步，创建order_product订单商品信息表信息，完善数据。第四步，创建order_student 订单学生关联关系（订单人员关系表），完善数据。第五步，创建order_info订单表，完善数据。总结，下单过程需要维护pro_product商品信息表、pro_product_type商品类型表、order_accom_apply、order_product订单商品信息表、order_student 订单学生关联关系表、order_info订单表六张表数据。
